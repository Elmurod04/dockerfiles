cd /opt/monitoring-lab/services/backend

# Build the image
docker build -t backend:local .

# Check image size
docker images backend:local

# Run it standalone (no compose)
docker run -d \
  --name backend-test \
  -p 3000:3000 \
  -e MONGO_URL=mongodb://localhost:27017/mydb \
  -e OTEL_EXPORTER_OTLP_ENDPOINT=http://localhost:4318 \
  backend:local

# Check if it's running
docker ps

# Check logs
docker logs backend-test

# Test the endpoint
curl http://localhost:3000/health

# Cleanup after test
docker stop backend-test && docker rm backend-test
