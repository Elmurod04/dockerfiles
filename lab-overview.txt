services:

  # ── DATABASE ────────────────────────────────────────────
  mongo:
    image: mongo:7
    container_name: mongo
    networks:
      - lab-net
    volumes:
      - mongo-data:/data/db
    healthcheck:
      test: ["CMD", "mongosh", "--eval", "db.adminCommand('ping')"]
      interval: 10s
      timeout: 5s
      retries: 10
      start_period: 20s
    restart: unless-stopped

  mongo-express:
    image: mongo-express:1.0.2
    container_name: mongo-express
    ports:
      - "8081:8081"
    environment:
      - ME_CONFIG_MONGODB_SERVER=mongo
      - ME_CONFIG_BASICAUTH=false
    networks:
      - lab-net
    depends_on:
      mongo:
        condition: service_healthy
    restart: unless-stopped

  # ── CUSTOM SERVICES ─────────────────────────────────────
  backend:
    build:
      context: ./services/backend
      dockerfile: Dockerfile
    container_name: backend
    extra_hosts:
      - "host.docker.internal:host-gateway"
    environment:
      - MONGO_URL=mongodb://mongo:27017/products
      - APP_NAME=backend
      - STATS_URL=http://stats:3001
      - OTEL_SERVICE_NAME=backend
      - OTEL_EXPORTER_OTLP_ENDPOINT=http://host.docker.internal:4318   # HTTP port matches protocol below
      - OTEL_EXPORTER_OTLP_PROTOCOL=http/protobuf                       # keeping your protocol choice
      - OTEL_TRACES_SAMPLER=parentbased_traceidratio
      - OTEL_TRACES_SAMPLER_ARG=1.0
      - OTEL_TRACES_EXPORTER=otlp
      - OTEL_PROPAGATORS=tracecontext,baggage
    networks:
      - lab-net
    depends_on:
      mongo:
        condition: service_healthy
      stats:
        condition: service_started
    healthcheck:
      test: ["CMD", "wget", "-qO-", "http://localhost:3000/health"]
      interval: 15s
      timeout: 5s
      retries: 8
      start_period: 40s
    restart: unless-stopped

  stats:
    build:
      context: ./services/stats
      dockerfile: Dockerfile
    container_name: stats
    extra_hosts:
      - "host.docker.internal:host-gateway"
    environment:
      - MONGO_URL=mongodb://mongo:27017/products
      - APP_NAME=stats
      - OTEL_SERVICE_NAME=stats
      - OTEL_EXPORTER_OTLP_ENDPOINT=http://host.docker.internal:4318   # HTTP port matches protocol
      - OTEL_EXPORTER_OTLP_PROTOCOL=http/protobuf
      - OTEL_TRACES_SAMPLER=parentbased_traceidratio
      - OTEL_TRACES_SAMPLER_ARG=1.0
      - OTEL_TRACES_EXPORTER=otlp
      - OTEL_PROPAGATORS=tracecontext,baggage
    networks:
      - lab-net
    depends_on:
      mongo:
        condition: service_healthy
    healthcheck:
      test: ["CMD", "wget", "-qO-", "http://localhost:3001/health"]
      interval: 15s
      timeout: 5s
      retries: 8
      start_period: 40s
    restart: unless-stopped

  # ── NGINX ────────────────────────────────────────────────
  nginx:
    build:
      context: ./services
      dockerfile: nginx/Dockerfile
    container_name: nginx
    ports:
      - "8080:80"
    networks:
      - lab-net
    depends_on:
      backend:
        condition: service_started
    restart: unless-stopped

  # ── OBSERVABILITY ────────────────────────────────────────
  grafana:
    image: grafana/grafana:11.4.0
    container_name: grafana
    ports:
      - "3000:3000"
    extra_hosts:
      - "host.docker.internal:host-gateway"
    environment:
      - GF_AUTH_ANONYMOUS_ENABLED=false
      - GF_AUTH_DISABLE_LOGIN_FORM=false
      - GF_SECURITY_ADMIN_USER=admin
      - GF_SECURITY_ADMIN_PASSWORD=${GRAFANA_ADMIN_PASSWORD:-changeme}
      - GF_USERS_ALLOW_SIGN_UP=false
      - GF_SECURITY_SECRET_KEY=${GRAFANA_SECRET_KEY:-change-this-secret}
      - GF_DASHBOARDS_DEFAULT_HOME_DASHBOARD_PATH=/etc/grafana/dashboards/lab-overview.json
    volumes:
      - ./monitoring/grafana/provisioning:/etc/grafana/provisioning:ro
      - ./monitoring/grafana/dashboards:/etc/grafana/dashboards:ro
      - grafana-data:/var/lib/grafana
    networks:
      - lab-net
    restart: unless-stopped

# ── NETWORK & VOLUMES ──────────────────────────────────────
networks:
  lab-net:
    driver: bridge

volumes:
  mongo-data:
  grafana-data:
